Configure DHCP Server

R1# conf t
R1(config)# ip dhcp pool LANPOOL
R1(dhcp-config)# network 192.168.1.0 255.255.255.0
R1(dhcp-config)# default-router 192.168.1.1
R1(dhcp-config)# exit
R1(config)# int f0/0
R1(config-if)# ip add 192.168.1.1 255.255.255.0
R1(config-if)# no sh
R1(config-if)# exit

PC1> dhcp
PC1> sh ip
